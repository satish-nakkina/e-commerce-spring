package com.assignment.service;

import com.assignment.dao.StateRepository;;
import com.assignment.entity.State;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Optional;

@Service
public class StateServiceImpl implements StateService {
    @Autowired
    StateRepository stateRepository;
    public static class DistrictNotFoundException extends RuntimeException {
        public DistrictNotFoundException(String message) {
            super(message);
        }
    }
    @Override
    public List<State> findAll() {
        return stateRepository.findAll();
    }

    @Override
    public State findById(int theId) {
        Optional<State> stateById=stateRepository.findById(theId);
        State state=null;
        if (stateById.isPresent())
        {
            state= stateById.get();
        }
        else{
            throw new DistrictNotFoundException("Did not find district id - " + theId);

        }
        return state;

    }

    @Override
    public void save(State theInstructor) {
        stateRepository.save(theInstructor);
    }

    @Override
    public void deleteById(int theId) {
        stateRepository.deleteById(theId);
    }
}
