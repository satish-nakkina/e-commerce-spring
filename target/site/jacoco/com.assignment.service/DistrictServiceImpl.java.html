package com.assignment.service;

import com.assignment.dao.DistrictRepository;
import com.assignment.entity.District;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import java.util.List;
import java.util.Optional;

@Service
public class DistrictServiceImpl implements  DistrictService{
    @Autowired
    DistrictRepository districtRepository;

    public static class StateNotFoundException extends RuntimeException {
        public StateNotFoundException(String message) {
            super(message);
        }
    }
    public List<District> findAll(){
        return  districtRepository.findAll();
    }

    @Override
    public void save(District theCourse) {
        districtRepository.save(theCourse);
    }

    @Override
    public District findById(int theId) {
        Optional<District> districtById=districtRepository.findById(theId);
        District district=null;
        if (districtById.isPresent())
        {
            district= districtById.get();
        }
        else{
            throw new StateNotFoundException("Did not find state id - " + theId);

        }
        return district;
    }
    @Override
    public void deleteById(int theId) {
        districtRepository.deleteById(theId);
    }

}
